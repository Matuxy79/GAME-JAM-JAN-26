# Input Map Configuration for Vampire Survival
# Add these input actions in Project Settings > Input Map

# Movement
move_up: W, Up Arrow
move_down: S, Down Arrow  
move_left: A, Left Arrow
move_right: D, Right Arrow

# Game Controls
pause: Escape
activate_powerup_choice: Space
fire: Left Mouse Button

# UI Controls
ui_accept: Enter, Space
ui_cancel: Escape
ui_select: Left Mouse Button
